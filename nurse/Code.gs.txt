// ── HTTP handlers ──────────────────────────────────────

function doGet(e) {
  try {
    const p = e.parameter || {};
    if (p.action === 'get') return ok(getOne(Number(p.row)));
    return ok(listAll());
  } catch(err) { return fail(err.message); }
}

function doPost(e) {
  try {
    const b = JSON.parse(e.postData.contents);
    if (b.action === 'add')      return ok(addRow(b.data));
    if (b.action === 'update')   return ok(updateRow(b.data));
    if (b.action === 'delete')   return ok(deleteRow(Number(b.row)));
    if (b.action === 'ai-parse') return ok(aiParse(b));
    return fail('unknown action');
  } catch(err) { return fail(err.message); }
}

// ── Response helpers ───────────────────────────────────

function ok(d) {
  return ContentService
    .createTextOutput(JSON.stringify({ ok: true, ...d }))
    .setMimeType(ContentService.MimeType.JSON);
}
function fail(msg) {
  return ContentService
    .createTextOutput(JSON.stringify({ ok: false, error: msg }))
    .setMimeType(ContentService.MimeType.JSON);
}

// ── Sheet helpers ──────────────────────────────────────

function sh() {
  return SpreadsheetApp.openById(SS_ID).getSheetByName(TAB_NAME);
}
function fmtDate(v) {
  if (!v) return '';
  if (v instanceof Date) return Utilities.formatDate(v, 'Asia/Taipei', 'yyyy/MM/dd');
  return String(v).trim();
}
function clean(v) {
  const s = String(v == null ? '' : v).trim();
  return (s === '#N/A' || s.toUpperCase() === 'N/A') ? '' : s;
}
function safeAge(v) {
  const n = parseInt(v, 10);
  return (!isNaN(n) && n >= 1 && n <= 120) ? n : null;
}

function rowToObj(row, rowNum) {
  const raw = String(row[C.面談日期] || '');
  const resigned = raw.includes('離職');
  return {
    _row:            rowNum,
    狀態:            resigned ? '已離職' : '在職',
    面談日期:        resigned ? '' : fmtDate(row[C.面談日期]),
    姓名:            clean(row[C.姓名]),
    員工編號:        String(row[C.員工編號] || ''),
    生日:            fmtDate(row[C.生日]),
    部門:            clean(row[C.部門]),
    性別:            clean(row[C.性別]),
    備註:            clean(row[C.備註]),
    人因:            clean(row[C.人因]),
    缺血風險數值:    clean(row[C.缺血風險數值]),
    缺血風險等級:    clean(row[C.缺血風險等級]),
    工作負荷程度:    clean(row[C.工作負荷程度]),
    異常工作負荷結果:clean(row[C.異常工作負荷]),
    代謝症候群:      clean(row[C.代謝症候群]),
    年齡:            safeAge(row[C.年齡]),
    中高齡:          clean(row[C.中高齡]),
    職護:            clean(row[C.職護]),
    醫生:            clean(row[C.醫生]),
    第一次面談:      fmtDate(row[C.第一次面談]),
    第二次面談:      fmtDate(row[C.第二次面談]),
  };
}

// ── CRUD ──────────────────────────────────────────────

function listAll() {
  const vals = sh().getDataRange().getValues();
  const data = [];
  for (let i = 1; i < vals.length; i++) data.push(rowToObj(vals[i], i + 1));
  return { count: data.length, data };
}

function getOne(rowNum) {
  const row = sh().getRange(rowNum, 1, 1, NCOLS).getValues()[0];
  return { data: rowToObj(row, rowNum) };
}

function objToRow(d) {
  // 已離職時寫入'離職'以保留標記（rowToObj 讀取時以此判斷狀態）
  const dateFld = (d.狀態 === '已離職') ? '離職' : (d.面談日期||'');
  return [
    dateFld, d.姓名||'',          d.員工編號||'',
    d.生日||'',     d.部門||'',           d.性別||'',
    d.備註||'',     d.人因||'',           d.缺血風險數值||'',
    d.缺血風險等級||'', d.工作負荷程度||'', d.異常工作負荷結果||'',
    d.代謝症候群||'', d.年齡||'',          d.中高齡||'',
    d.職護||'',     d.醫生||'',           d.第一次面談||'',
    d.第二次面談||'',
  ];
}

function addRow(d) {
  sh().appendRow(objToRow(d));
  return { message: '新增成功' };
}

function updateRow(d) {
  if (!d._row) throw new Error('缺少 _row');
  sh().getRange(d._row, 1, 1, NCOLS).setValues([objToRow(d)]);
  return { message: '更新成功' };
}

function deleteRow(rowNum) {
  if (!rowNum || rowNum < 2) throw new Error('無效的列號');
  sh().deleteRow(rowNum);
  return { message: '已刪除' };
}

function aiParse(b) {
  // 支援多檔：b.files = [{fileData, fileType, fileName}, ...]
  // 向下相容舊版單檔格式：b.fileData / b.fileType
  const fileList = b.files && b.files.length
    ? b.files
    : (b.fileData ? [{ fileData: b.fileData, fileType: b.fileType, fileName: b.fileName }] : []);

  if (!fileList.length) throw new Error('未提供任何檔案');

  const images = fileList.map(f => {
    const ft = f.fileType || '';
    if (!ft.startsWith('image/') && ft !== 'application/pdf')
      throw new Error('僅支援 PDF 及圖片格式（JPG、PNG）：' + ft);
    return 'data:' + ft + ';base64,' + f.fileData;
  });

  // 若前端帶入檔名解析的員工編號 / 姓名，附加為 Prompt 提示
  let prompt = PARSE_PROMPT;
  if (b.name || b.empNo) {
    prompt += '\n\n【人資系統檔名資訊（請以此為準）】\n'
      + (b.empNo ? '員工編號：' + b.empNo + '\n' : '')
      + (b.name  ? '姓名：' + b.name + '（直接採用此姓名填入，不需從報告重新判讀）\n' : '');
  }

  // 呼叫公司 IT OpenRouter Dispatch API（Gemini 2.5 Flash）
  const resp = UrlFetchApp.fetch(IT_URL, {
    method: 'post',
    headers: {
      'Content-Type':   'application/json',
      'X-SDK-Key':      IT_SDK_KEY,
      'X-User-Token':   IT_TOKEN,
      'X-Project-Code': IT_PROJECT,
    },
    payload: JSON.stringify({
      model:  IT_MODEL,
      text:   prompt,
      images: images,
    }),
    muteHttpExceptions: true,
  });

  const respJson = JSON.parse(resp.getContentText());
  if (respJson.error) throw new Error('AI API：' + (respJson.error.message || respJson.error));

  // 相容多種回傳格式
  const text = (
    respJson.data ||
    (respJson.choices && respJson.choices[0] && respJson.choices[0].message && respJson.choices[0].message.content) ||
    respJson.text ||
    ''
  ).trim();

  const match = text.match(/\{[\s\S]*\}/);
  if (!match) throw new Error('AI 未回傳有效 JSON，請重試');

  const result = JSON.parse(match[0]);
  return { result };
}
